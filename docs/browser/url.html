The browser/url module

API

import {
  to_blob_url,
  resolve_blob_url,
} from 'https://m.bednarz.dev/browser/url.js';
to_blob_url(data: string, type: string): string
  • create a blob from data and type
  • create and return an object URL from that blob 
resolve_blob_url(
  resolve: (url: string) => Promise<any>,
  data: string,
  type: string
): Promise<any >
  • call to_blob_url with data and type
  • call resolve with the returned URL
  • revoke the object url when resolve is settled
  • return the settled promise
